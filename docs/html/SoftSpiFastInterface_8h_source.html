41 template <uint8_t T_LATCH_PIN, uint8_t T_DATA_PIN, uint8_t T_CLOCK_PIN>

47  void begin() const {

51  }

54  void end() const {
55  pinModeFast(T_LATCH_PIN, INPUT);
56  pinModeFast(T_DATA_PIN, INPUT);
57  pinModeFast(T_CLOCK_PIN, INPUT);

61  void send8(uint8_t value) const {
62  digitalWriteFast(T_LATCH_PIN, LOW);
63  shiftOutFast(value);
64  digitalWriteFast(T_LATCH_PIN, HIGH);
65  }
66 
68  void send16(uint16_t value) const {
69  uint8_t msb = (value & 0xff00) >> 8;
70  uint8_t lsb = (value & 0xff);
71  send16(msb, lsb);
72  }
73 
75  void send16(uint8_t msb, uint8_t lsb) const {
76  digitalWriteFast(T_LATCH_PIN, LOW);
77  shiftOutFast(msb);
78  shiftOutFast(lsb);
79  digitalWriteFast(T_LATCH_PIN, HIGH);
80  }
81 
82  private:
83  static void shiftOutFast(uint8_t output) {
84  uint8_t mask = 0x80; // start with the MSB
85  for (uint8_t i = 0; i < 8; i++) {
86  digitalWriteFast(T_CLOCK_PIN, LOW);
87  if (output & mask) {
88  digitalWriteFast(T_DATA_PIN, HIGH);
89  } else {
90  digitalWriteFast(T_DATA_PIN, LOW);
91  }
92  digitalWriteFast(T_CLOCK_PIN, HIGH);
93  mask >>= 1;
94  }
95  }
96 };
